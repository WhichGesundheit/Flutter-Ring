-- 1. Create the Users/Profiles Table (Meta-progression)
CREATE TABLE IF NOT EXISTS public.users (
    id UUID REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY,
    username TEXT,
    gold_total INT DEFAULT 0,
    unlocked_item_ids TEXT[] DEFAULT '{}'::TEXT[]
);

-- Enable Row Level Security (RLS) for Users
ALTER TABLE public.users ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow individual read/write to own profile" 
ON public.users FOR ALL TO authenticated USING (auth.uid() = id);

-- 2. Create the Game Runs Table (Active Run Tracker)
CREATE TABLE IF NOT EXISTS public.game_runs (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id UUID REFERENCES public.users(id) ON DELETE CASCADE NOT NULL,
    current_day INT DEFAULT 3,
    player_hp INT DEFAULT 60,
    player_max_hp INT DEFAULT 60,
    status TEXT DEFAULT 'active' CHECK (status IN ('active', 'won', 'lost')),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL
);

-- Enable RLS for Game Runs
ALTER TABLE public.game_runs ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow players to manage their own game runs" 
ON public.game_runs FOR ALL TO authenticated USING (auth.uid() = user_id);

-- 3. Create the Run Inventory Table (Active Items Slots)
CREATE TABLE IF NOT EXISTS public.run_inventory (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    run_id UUID REFERENCES public.game_runs(id) ON DELETE CASCADE NOT NULL,
    item_id TEXT NOT NULL,
    slot_position INT NOT NULL
);

-- Enable RLS for Run Inventory
ALTER TABLE public.run_inventory ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow players to manage items for their runs" 
ON public.run_inventory FOR ALL TO authenticated USING (
    EXISTS (
        SELECT 1 FROM public.game_runs 
        WHERE public.game_runs.id = public.run_inventory.run_id 
        AND public.game_runs.user_id = auth.uid()
    )
);

-- 4. Automatic Trigger: Creates a public.users row whenever a new auth user logs in anonymously
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.users (id, username, gold_total)
  VALUES (new.id, 'Survivor_' || substr(new.id::text, 1, 5), 0);
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
